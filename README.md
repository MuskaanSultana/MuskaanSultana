name: Generate Snake Animation

on:

  push:
    branches:
      - main

jobs:
  generate:
    permissions:
      contents: write
    runs-on: ubuntu-latest
    timeout-minutes: 5

        with:
          target_branch: output
          build_dir: dist

          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
